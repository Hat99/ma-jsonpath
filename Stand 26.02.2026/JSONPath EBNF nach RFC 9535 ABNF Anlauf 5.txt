//JSONPath EBNF nach RFC 9535 ABNF (Anlauf 5)//

/NOTE: ... -> ABNF enthält detaillierte Character-Definitionen, die hier später auch integriert werden müssen!/
/NOTE: whitespaces werden nicht explizit beachtet, dies muss noch auf Korrektheit geprüft werden!/
/NOTE: nichtterminale werden klein geschrieben, TERMINALE groß/
/NOTE: , wird weggelassen, dafür werden Klammern penibler gesetzt (hoffentlich...)/
/NOTE: white space edge case: "$" ist erlaubt, "$ " nicht?/



jsonpath-query	=	ROOT {seg}

ROOT		=	"$"

seg		=	childseg | descseg

childseg 	=	bracketed | ("." (WILDCARD | member-name-shorthand))

descseg 	=	".." (bracketed | WILDCARD | member-name-shorthand)

bracketed	=	"[" selector {"," selector} "]"





selector 	=	name-selector | WILDCARD-selector | slice-selector | index-selector | filter-selector



name-selector 	=	('"' {double-quoted} '"') | ("'" {single-quoted} "'")

double-quoted 	=	unescaped | "'" | (ESC '"') | (ESC escapable)

single-quoted	=	unescaped | '"' | (ESC "'") | (ESC escapable)

ESC		=	"\"

unescaped	=	[a-zA-Z0-9] | " " | ...								// vereinfachte Fassung, *NICHT* RFC konform!

escapable	=	...



member-name-shorthand =	name-first {name-char}

name-first	=	ALPHA | "_" | ...

name-char	=	name-first | DIGIT

ALPHA		=	[a-zA-Z]

DIGIT		=	[0-9]



WILDCARD-selector =	"*"



slice-selector	=	[start] ":" [end] [":" [step]]							// -> nur ":" erlaubt? wenn ja, [:] äquivalent zu [*]?

start		=	int

end		=	int

step		=	int

int		=	"0" | (["-"] DIGIT1 {DIGIT})

DIGIT1		=	[1-9]



index-selector	=	int



filter-selector	=	"?" logical-expr

logical-expr	=	logical-or-expr

logical-or-expr =	logical-and-expr {"||" logical-and-expr}

logical-and-expr=	basic-expr {"&&" basic-expr}

basic-expr	=	paren-expr | comparison-expr | test-expr

paren-expr	=	[LOGICAL-NOT-OP] "(" logical-expr ")"						//ermöglicht Klammerung

LOGICAL-NOT-OP	=	"!"

test-expr	=	[LOGICAL-NOT-OP] (filter-query | function-expr)

filter-query	=	rel-query | jsonpath-query

rel-query	=	CURRENT {seg}

CURRENT		=	"@"

comparison-expr =	comparable comparison-op comparable

literal		=	number | string-literal | TRUE | FALSE | NULL

comparable	=	literal | singular-query | function-expr

comparison-op	=	"==" | "!=" | "<=" | ">=" | "<" | ">"

singular-query 	=	rel-singular-query | abs-singular-query

rel-singular-query =	CURRENT singular-query-segments

abs-singular-query = 	ROOT singular-query-segments

singular-query-segments={name-segment | index-segment}

name-segment 	=	("[" name-selector "]") | ("." member-name-shorthand)

index-segment	=	"[" index-selector "]"



number		=	(int | "-0") [frac] [exp]

frac		=	"." DIGIT {DIGIT}

exp		= 	"e" ["-" | "+"] DIGIT {DIGIT}

TRUE		=	"true"										//true, false und null müssen lowercase sein!

FALSE		=	"false"

NULL		=	"null"



function-name	=	function-name-first {function-name-char}

function-name-first = 	LCALPHA

function-name-char =	function-name-first | "_" DIGIT

LCALPHA		=	[a-z]										//Lower Case ALPHAbet

function-expr	=	function-name "(" [function-argument {"," function-argument}] ")"		//prüfen, ob Funktionsdefinition vorhanden muss noch passieren

function-argument = 	literal | filter-query | logical-expr | function-expr
